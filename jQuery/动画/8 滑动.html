1、下拉 语法： slideDown([speed,[easing],[fn]]) 例： $(function(){ $("button").eq(0).click(function(){ $("div").slideDown(); }) }) 2、上拉 语法： slideUp([speed,[easing],[fn]]) 例： $(function(){ $("button").eq(1).click(function(){ $("div").slideUp(500); }) }) 3、切换 语法： slideToggle([speed,[easing],[fn]]) 例： $(function(){ $("button").eq(2).click(function(){ $("div").slideToggle(500) }) })
